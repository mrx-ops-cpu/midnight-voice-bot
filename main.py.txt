import discord
from discord.ext import commands
from flask import Flask
from threading import Thread

# 1. Створюємо міні-сайт всередині бота. 
# Це потрібно, щоб хостинг Render не вимикав бота за "неактивність".
app = Flask('')

@app.route('/')
def home():
    return "MIDNIGHT BOT IS ONLINE"

def run():
    app.run(host='0.0.0.0', port=8080)

def keep_alive():
    t = Thread(target=run)
    t.start()

# 2. Основні налаштування бота
intents = discord.Intents.default()
intents.voice_states = True  # Дозволяємо боту бачити голосові канали
intents.message_content = True 

bot = commands.Bot(command_prefix='!', intents=intents)

# ВСТАВ СЮДИ ID СВОГО КАНАЛУ (той, що ти копіював через ПК)
VOICE_ID = 1458906259922354277 

@bot.event
async def on_ready():
    print(f'Авторизовано як: {bot.user.name}')
    channel = bot.get_channel(VOICE_ID)
    if channel:
        await channel.connect()
        print(f'Бот зайшов у канал: {channel.name}')

# Запускаємо веб-сервер та бота
keep_alive()
